// --- MATCHES & CHAT LOGIC --- async function loadMatches() { // Defaulting to the global activeSport variable await fetchPredictions(activeSport); } async function fetchPredictions(sport, tier = 'normal') { const container = document.getElementById(sport + '-matches'); if (container) { container.innerHTML = '

Loading predictions...

'; } try { // FIXED: Now uses window.API_BASE_URL (Render) and properly defined tier/sport variables const response = await fetch(`${window.API_BASE_URL}/api/predictions?tier=${tier}&sport=${sport}`, { method: 'GET', headers: { 'x-api-key': 'skcs_user_12345', 'Content-Type': 'application/json' } }); if (!response.ok) throw new Error('API Error'); const data = await response.json(); // Render the received data to the UI renderSport(sport, sport + '-matches', data.predictions || []); // Update global tracking matchesLoaded = true; if (data.predictions) { skcsMatches = [...skcsMatches, ...data.predictions.flatMap(p => p.matches)]; } } catch (error) { console.error("Fetch error:", error); if (container) { container.innerHTML = '

Connection Error: Ensure backend is running.

'; } } } function renderSport(sport, containerId, predictions) { const container = document.getElementById(containerId); if (!container) return; container.innerHTML = predictions.length === 0 ? '

No predictions available.

' : ''; predictions.forEach(p => { const card = document.createElement('div'); card.className = 'prediction-card'; const isAcca = p.matches && p.matches.length > 1; const accaBadge = isAcca ? 'ACCUMULATOR' : ''; let matchesHtml = ""; if (p.matches && p.matches.length > 0) { matchesHtml = isAcca ? p.matches.map(m => `${m.home_team} vs ${m.away_team}`).join('
') : `${p.matches[0].home_team} vs ${p.matches[0].away_team}`; } const predictionText = p.matches && p.matches[0] ? p.matches[0].prediction.replace('_',' ') : 'N/A'; card.innerHTML = ` ${accaBadge}
${matchesHtml}
${sport.toUpperCase()}
${predictionText}
${p.total_confidence || 0}% Confidence
`; container.appendChild(card); }); } async function sendMessage() { const inputField = document.getElementById("chat-input"); const msg = inputField.value.trim(); if (!msg) return; inputField.value = ""; addMessage(msg, "user"); try { // FIXED: Points to Render API for Chat const response = await fetch(`${window.API_BASE_URL}/api/chat`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': 'skcs_user_12345' }, body: JSON.stringify({ message: msg }) }); const data = await response.json(); addMessage(data.response || "No response from AI", "bot"); } catch (error) { addMessage("Connection Error: Unable to reach the SKCS Brain.", "error"); } } function addMessage(text, type = "bot") { const box = document.getElementById("chat-messages"); const div = document.createElement("div"); div.className = `chat-bubble ${type}`; div.innerText = text; box.appendChild(div); box.scrollTop = box.scrollHeight; } document.addEventListener('DOMContentLoaded', () => { const sendBtn = document.getElementById('chatbot-send-btn'); if(sendBtn) sendBtn.addEventListener('click', sendMessage); const chatInput = document.getElementById('chat-input'); if (chatInput) { chatInput.addEventListener('keypress', function(e) { if (e.key === 'Enter') sendMessage(); }); } document.querySelectorAll('.tab-button').forEach(btn => { btn.addEventListener('click', function() { const tid = this.getAttribute('data-tab'); activeSport = tid; document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active')); document.querySelectorAll('.tab-button').forEach(b => b.classList.remove('active')); const activeContent = document.getElementById(tid); if (activeContent) activeContent.classList.add('active'); this.classList.add('active'); // Trigger fetch for the newly selected sport fetchPredictions(activeSport); }); }); // Initial startup if (typeof checkSession === 'function') checkSession(); loadMatches(); }); const scrollTopBtn = document.getElementById('scrollTopBtn'); if (scrollTopBtn) { window.addEventListener('scroll', () => { scrollTopBtn.classList.toggle('show', window.scrollY > 300); }); scrollTopBtn.addEventListener('click', () => window.scrollTo({ top: 0, behavior: 'smooth' }));
